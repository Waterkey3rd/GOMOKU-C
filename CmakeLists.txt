cmake_minimum_required(VERSION 3.22)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS ON)

if(NOT CMAKE_BUILD_TYPE)
    set(CMAKE_BUILD_TYPE "Debug")
endif()

set(MY_PROJECT_NAME "GomokuGame")
project(${MY_PROJECT_NAME} LANGUAGES CXX)

set(CMAKE_EXPORT_COMPILE_COMMANDS TRUE)
message("Build type: " ${CMAKE_BUILD_TYPE})

file(GLOB_RECURSE APP_SRH ${CMAKE_CURRENT_SOURCE_DIR}/App/*.h)
file(GLOB_RECURSE APP_SRCPP ${CMAKE_CURRENT_SOURCE_DIR}/App/*.cpp)

set(RAYLIB_PATH "${CMAKE_CURRENT_SOURCE_DIR}/include/raylib")

add_executable(${MY_PROJECT_NAME} ${APP_SRCPP} ${APP_SRH})

target_include_directories(${MY_PROJECT_NAME} PRIVATE 
    ${CMAKE_CURRENT_SOURCE_DIR}/App
    ${RAYLIB_PATH}/include
)

target_link_directories(GomokuGame PRIVATE 
    ${RAYLIB_PATH}/lib
)

target_link_libraries(GomokuGame PRIVATE 
    raylib 
    opengl32 
    gdi32 
    winmm
)

if(WIN32 AND CMAKE_BUILD_TYPE STREQUAL "Release")
    set_property(TARGET ${MY_PROJECT_NAME} PROPERTY WIN32_EXECUTABLE ON)
endif()
